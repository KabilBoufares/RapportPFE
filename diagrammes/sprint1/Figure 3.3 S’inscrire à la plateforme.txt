@startuml
actor Visiteur
boundary "Application mobile" as App
control "Back-End FESTY" as Backend
database "Base de données" as DB
participant "Service OTP/SMS" as SMS

autonumber

' Demande du formulaire
Visiteur -> App : Demander le formulaire d'inscription
activate App
App --> Visiteur : Afficher le formulaire
deactivate App

' Soumission des informations
Visiteur -> App : Soumettre les informations d'inscription
activate App
App -> Backend : Envoyer les informations
deactivate App

activate Backend
Backend -> Backend : Valider les champs saisis
alt informations invalides
    Backend --> App : Message d'erreur (E1)
    deactivate Backend
    activate App
    App --> Visiteur : Afficher l'erreur
    deactivate App
else informations valides
    Backend -> DB : Vérifier l'unicité de l'email
    deactivate Backend
    activate DB
    DB --> Backend : Email disponible / indisponible
    deactivate DB
    alt email déjà utilisé
        activate Backend
        Backend --> App : Message d'erreur (E2)
        deactivate Backend
        activate App
        App --> Visiteur : Afficher l'erreur
        deactivate App
    else email disponible
        activate Backend
        Backend -> DB : Créer le compte utilisateur (statut en attente)
        deactivate Backend
        activate DB
        DB --> Backend : Confirmation création
        deactivate DB
        activate Backend
        Backend -> Backend : Générer le code OTP
        Backend -> DB : Enregistrer le code OTP
        deactivate Backend
        activate DB
        DB --> Backend : Code enregistré
        deactivate DB
        activate Backend
        Backend -> SMS : Envoyer le code OTP (au numéro)
        deactivate Backend
        activate SMS
        SMS --> Backend : Confirmation d'envoi
        deactivate SMS
        activate Backend
        Backend --> App : Demander la saisie du code OTP
        deactivate Backend
        activate App
        App --> Visiteur : Afficher le champ de saisie OTP
        deactivate App

        ' Saisie du code OTP
        Visiteur -> App : Saisir le code OTP
        activate App
        App -> Backend : Envoyer le code OTP saisi
        deactivate App
        activate Backend
        Backend -> DB : Vérifier le code OTP
        deactivate Backend
        activate DB
        DB --> Backend : Code valide / invalide
        deactivate DB
        alt code OTP incorrect ou expiré
            activate Backend
            Backend --> App : Message d'erreur (E3)
            deactivate Backend
            activate App
            App --> Visiteur : Afficher l'erreur
            deactivate App
        else code OTP valide
            activate Backend
            Backend -> DB : Activer le compte (statut actif)
            deactivate Backend
            activate DB
            DB --> Backend : Confirmation activation
            deactivate DB
            activate Backend
            Backend --> App : Confirmation d'inscription réussie
            deactivate Backend
            activate App
            App --> Visiteur : Afficher le succès
            deactivate App
        end
    end
end
@enduml
