@startuml
actor Utilisateur
participant "Application mobile" as App
participant "Back-End FESTY" as System
database "Base de données" as DB

autonumber

Utilisateur -> App : Accéder à la page de profil
activate App

App -> System : Demander les informations du profil
activate System
System -> DB : Récupérer les informations du profil
activate DB
DB --> System : Informations du profil
deactivate DB

System --> App : Envoyer les données du profil
deactivate System
App --> Utilisateur : Afficher les informations du profil
deactivate App

alt quitter sans enregistrer
    Utilisateur -> App : Quitter la page sans enregistrer
    activate App
    App --> Utilisateur : Fin de l'interaction
    deactivate App
else modifier les informations
    Utilisateur -> App : Modifier les informations souhaitées
    activate App
    Utilisateur -> App : Soumettre les modifications
    App -> System : Envoyer les nouvelles données
    activate System

    System -> System : Valider les données saisies

    alt données invalides
        System --> App : Message d'erreur (E1)
        App --> Utilisateur : Afficher "Données invalides"
    else données valides
        System -> DB : Mettre à jour les informations du profil
        activate DB
        DB --> System : Confirmation de mise à jour
        deactivate DB
        System --> App : Message de confirmation
        App --> Utilisateur : Afficher "Profil mis à jour avec succès"
    end

    deactivate System
    deactivate App
end
@enduml
