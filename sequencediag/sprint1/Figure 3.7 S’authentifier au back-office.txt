@startuml
actor Administrateur
participant "Interface d'administration" as AdminUI
participant "Back-End FESTY" as System
database "Base de données" as DB

autonumber

' Étape 1 – Accès à l'interface
Administrateur -> AdminUI : Accéder à l'interface de connexion du back-office
activate AdminUI
AdminUI --> Administrateur : Afficher le formulaire de connexion
deactivate AdminUI

' Étape 2 – Saisie des identifiants
Administrateur -> AdminUI : Saisir l'email et le mot de passe
activate AdminUI
AdminUI -> System : Envoyer les identifiants (email, mot de passe)
activate System

' Étape 3 – Vérification des identifiants
System -> DB : Vérifier l'existence de l'email
activate DB
DB --> System : Email trouvé / introuvable
deactivate DB

alt email introuvable
    System --> AdminUI : Message d'erreur (E1)
    AdminUI --> Administrateur : Afficher "Identifiants incorrects"
    deactivate System
    deactivate AdminUI

else email trouvé
    System -> DB : Récupérer les informations du compte
    activate DB
    DB --> System : Informations du compte (mot de passe haché, rôle)
    deactivate DB

    System -> System : Vérifier le mot de passe

    alt mot de passe incorrect
        System --> AdminUI : Message d'erreur (E1)
        AdminUI --> Administrateur : Afficher "Identifiants incorrects"
        deactivate System
        deactivate AdminUI

    else mot de passe correct
        ' Étape 4 – Vérification du rôle administrateur
        System -> System : Vérifier le rôle administrateur

        ' Étape 5 – Demande du code 2FA
        System --> AdminUI : Demander le code 2FA
        deactivate System
        AdminUI --> Administrateur : Afficher le champ de saisie du code 2FA
        deactivate AdminUI

        ' Étape 6 – Saisie et vérification du code 2FA
        Administrateur -> AdminUI : Saisir le code 2FA
        activate AdminUI
        AdminUI -> System : Envoyer le code 2FA saisi
        activate System

        System -> DB : Récupérer la configuration 2FA
        activate DB
        DB --> System : Configuration 2FA (secret)
        deactivate DB

        System -> System : Vérifier le code 2FA

        alt code 2FA invalide ou expiré
            System --> AdminUI : Message d'erreur (E2)
            AdminUI --> Administrateur : Afficher "Code 2FA invalide ou expiré"
        else code 2FA valide
            System -> System : Créer une session administrateur sécurisée
            System --> AdminUI : Authentification réussie
            AdminUI --> Administrateur : Rediriger vers l'espace d'administration
        end

        deactivate System
        deactivate AdminUI
    end
end
@enduml
