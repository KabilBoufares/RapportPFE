@startuml
actor Utilisateur
participant "Application mobile" as App
participant "Back-End FESTY" as System
database "Base de données" as DB

autonumber

' Étape 1 – Accès à l'interface
Utilisateur -> App : Accéder à l'interface de connexion
activate App
App --> Utilisateur : Afficher le formulaire de connexion
deactivate App

' Étape 2 – Saisie des identifiants
Utilisateur -> App : Saisir l'email et le mot de passe
activate App
App -> System : Envoyer les identifiants (email, mot de passe)
activate System

' Étape 3 – Vérification de l'email
System -> DB : Vérifier l'existence de l'email
activate DB
DB --> System : Email trouvé / introuvable
deactivate DB

alt email introuvable
    System --> App : Message d'erreur (E1)
    App --> Utilisateur : Afficher "Email introuvable"
else email trouvé
    ' Suite étape 3 – Récupération des informations du compte
    System -> DB : Récupérer les informations du compte
    activate DB
    DB --> System : Retourner les informations du compte
    deactivate DB

    ' Vérification du mot de passe côté Back-End
    System -> System : Vérifier le mot de passe

    alt mot de passe incorrect
        System --> App : Message d'erreur (E2)
        App --> Utilisateur : Afficher "Mot de passe incorrect"
    else mot de passe correct
        ' Étape 5 – Accès autorisé
        System -> System : Créer une session sécurisée
        System --> App : Authentification réussie
        App --> Utilisateur : Rediriger vers l'espace personnel
    end
end

deactivate System
deactivate App
@enduml
