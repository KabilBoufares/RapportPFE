@startuml
actor Visiteur
participant "Système" as System
database "Base de données" as DB
participant "Service SMS" as SMS

autonumber

Visiteur -> System : Demande le formulaire d'inscription
activate System
System --> Visiteur : Affiche le formulaire
deactivate System

Visiteur -> System : Soumet les informations\n(nom, email, mot de passe, téléphone)
activate System

System -> System : Valide les champs saisis
alt informations invalides
    System --> Visiteur : Message d'erreur (E1)
else informations valides
    System -> DB : Vérifier l'unicité de l'email
    activate DB
    DB --> System : Email disponible / indisponible
    deactivate DB
    alt email déjà utilisé
        System --> Visiteur : Message d'erreur (E2)
    else email disponible
        System -> DB : Créer le compte utilisateur (statut inactif)
        activate DB
        DB --> System : Confirmation
        deactivate DB
        System -> SMS : Demander l'envoi du code OTP\n(au numéro de téléphone)
        activate SMS
        SMS --> System : OTP envoyé
        deactivate SMS
        System --> Visiteur : Demander la saisie du code OTP
        deactivate System

        Visiteur -> System : Saisit le code OTP
        activate System
        System -> SMS : Vérifier le code OTP
        activate SMS
        SMS --> System : Code valide / invalide
        deactivate SMS
        alt code OTP incorrect ou expiré
            System --> Visiteur : Message d'erreur (E3)
        else code OTP valide
            System -> DB : Activer le compte (mettre à jour statut)
            activate DB
            DB --> System : Confirmation
            deactivate DB
            System --> Visiteur : Confirmation de l'inscription réussie
        end
        deactivate System
    end
end
@enduml
