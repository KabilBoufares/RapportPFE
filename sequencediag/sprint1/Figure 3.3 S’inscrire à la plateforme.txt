@startuml
actor Visiteur
participant "Application mobile" as App
participant "Système" as System
database "Base de données" as DB
participant "Service OTP/SMS" as SMS

autonumber

' Demande du formulaire
Visiteur -> App : Demander le formulaire d'inscription
activate App
App --> Visiteur : Afficher le formulaire
deactivate App

' Soumission des informations
Visiteur -> App : Soumettre les informations d'inscription
activate App
App -> System : Envoyer les informations
activate System

' Validation des champs
System -> System : Valider les champs saisis
alt informations invalides
    System --> App : Message d'erreur (E1)
    App --> Visiteur : Afficher l'erreur
else informations valides
    ' Vérification unicité email
    System -> DB : Vérifier l'unicité de l'email
    activate DB
    DB --> System : Email disponible / indisponible
    deactivate DB

    alt email déjà utilisé
        System --> App : Message d'erreur (E2)
        App --> Visiteur : Afficher l'erreur
    else email disponible
        ' Création du compte
        System -> DB : Créer le compte utilisateur (statut en attente)
        activate DB
        DB --> System : Confirmation création
        deactivate DB

        ' Génération et enregistrement du code OTP
        System -> System : Générer le code OTP
        System -> DB : Enregistrer le code OTP
        activate DB
        DB --> System : Code enregistré
        deactivate DB

        ' Envoi du code OTP par SMS
        System -> SMS : Envoyer le code OTP (au numéro)
        activate SMS
        SMS --> System : Confirmation d'envoi
        deactivate SMS

        ' Demande de saisie à l'utilisateur
        System --> App : Demander la saisie du code OTP
        deactivate System
        App --> Visiteur : Afficher le champ de saisie OTP
        deactivate App

        ' Saisie du code OTP
        Visiteur -> App : Saisir le code OTP
        activate App
        App -> System : Envoyer le code OTP saisi
        activate System

        ' Vérification du code OTP via la base de données
        System -> DB : Vérifier le code OTP
        activate DB
        DB --> System : Code valide / invalide
        deactivate DB

        alt code OTP incorrect ou expiré
            System --> App : Message d'erreur (E3)
            App --> Visiteur : Afficher l'erreur
        else code OTP valide
            ' Activation du compte
            System -> DB : Activer le compte (statut actif)
            activate DB
            DB --> System : Confirmation activation
            deactivate DB

            ' Confirmation finale
            System --> App : Confirmation d'inscription réussie
            App --> Visiteur : Afficher le succès
        end
        deactivate System
        deactivate App
    end
end
@enduml
