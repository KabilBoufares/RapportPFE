@startuml
actor Utilisateur
boundary "Application mobile" as App
control "Back-End FESTY" as Backend
database "Base de données" as DB

autonumber

' Étape 1 – Accès à l'espace de personnalisation
Utilisateur -> App : Accéder à l'espace de personnalisation
activate App
App -> Backend : Demander les catégories et centres d'intérêt
deactivate App

activate Backend
Backend -> DB : Récupérer les catégories et centres d'intérêt
deactivate Backend

activate DB
DB --> Backend : Liste des options
deactivate DB

activate Backend
Backend --> App : Envoyer les options
deactivate Backend

activate App
App --> Utilisateur : Afficher les catégories et centres d'intérêt disponibles
deactivate App

' Étape 2 – Sélection et soumission des préférences
Utilisateur -> App : Sélectionner ses préférences et soumettre
activate App
App -> Backend : Envoyer les préférences sélectionnées
deactivate App

activate Backend
Backend -> Backend : Vérifier que des préférences sont présentes

alt aucune préférence sélectionnée (E1)

    Backend --> App : Message d'erreur (E1)
    deactivate Backend

    activate App
    App --> Utilisateur : Afficher "Veuillez sélectionner au moins une préférence"
    deactivate App

else préférences présentes

    Backend -> DB : Enregistrer les préférences utilisateur
    deactivate Backend

    activate DB

    alt échec de l'enregistrement (E2)

        DB --> Backend : Erreur
        deactivate DB

        activate Backend
        Backend --> App : Message d'erreur (E2)
        deactivate Backend

        activate App
        App --> Utilisateur : Afficher "L'enregistrement des préférences a échoué"
        deactivate App

    else enregistrement réussi

        DB --> Backend : Confirmation
        deactivate DB

        activate Backend
        Backend -> Backend : Générer les recommandations selon les préférences

        Backend -> DB : Récupérer les événements recommandés
        deactivate Backend

        activate DB
        DB --> Backend : Liste des événements recommandés
        deactivate DB

        activate Backend
        Backend --> App : Envoyer les recommandations
        deactivate Backend

        activate App
        App --> Utilisateur : Afficher les recommandations adaptées
        deactivate App

    end
end
@enduml
