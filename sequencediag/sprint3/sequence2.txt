@startuml
actor Utilisateur
boundary "Application mobile" as App
control "Back-End FESTY" as Backend
database "Base de données" as DB

autonumber

' Étape 1 – Consultation du détail de l’événement
Utilisateur -> App : Demander le détail d'un événement
activate App
App -> Backend : Récupérer les détails de l'événement
deactivate App

activate Backend
Backend -> DB : Récupérer les informations de l'événement
deactivate Backend

activate DB
DB --> Backend : Détails de l'événement
deactivate DB

activate Backend
Backend --> App : Envoyer les détails
deactivate Backend

activate App
App --> Utilisateur : Afficher le détail de l'événement
deactivate App

' Étape 2 – Ouverture du chat de l’événement
Utilisateur -> App : Ouvrir le chat de l'événement
activate App
App -> Backend : Rejoindre le chat de l'événement
deactivate App

activate Backend
Backend -> DB : Vérifier ou enregistrer l'accès au chat
deactivate Backend

activate DB
DB --> Backend : Accès au chat confirmé
deactivate DB

activate Backend
Backend --> App : Confirmation d'accès au chat
deactivate Backend

activate App
App -> Backend : Charger les messages du chat
deactivate App

activate Backend
Backend -> DB : Récupérer les messages actifs
deactivate Backend

activate DB
DB --> Backend : Liste des messages
deactivate DB

activate Backend
Backend --> App : Envoyer les messages
deactivate Backend

activate App
App --> Utilisateur : Afficher l'interface de chat avec les messages existants
deactivate App

' Étape 3 – Saisie et envoi du message
Utilisateur -> App : Saisir un message et l'envoyer
activate App
App -> App : Vérifier que le message n'est pas vide

alt message vide (E1)
    App --> Utilisateur : Afficher "Le message ne peut pas être vide"
    deactivate App
else message valide
    App -> Backend : Envoyer le message
    deactivate App

    activate Backend
    Backend -> Backend : Vérifier l'accès au chat et le contenu

    alt utilisateur non autorisé (E3)
        Backend --> App : Retourner une erreur d'autorisation
        deactivate Backend
        activate App
        App --> Utilisateur : Afficher "Vous n'êtes pas autorisé à envoyer ce message"
        deactivate App
    else utilisateur autorisé
        Backend -> DB : Enregistrer le message
        deactivate Backend
        activate DB

        alt échec de l'enregistrement (E2)
            DB --> Backend : Erreur
            deactivate DB
            activate Backend
            Backend --> App : Retourner une erreur d'envoi
            deactivate Backend
            activate App
            App --> Utilisateur : Afficher "L'envoi du message a échoué"
            deactivate App
        else enregistrement réussi
            DB --> Backend : Message enregistré
            deactivate DB
            activate Backend
            Backend --> App : Retourner le message enregistré
            deactivate Backend
            activate App
            App --> Utilisateur : Afficher le message dans le chat
            deactivate App
        end
    end
end

@enduml
