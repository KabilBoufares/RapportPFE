@startuml
actor Partenaire
participant "Interface partenaire" as PartnerUI
participant "Back-End FESTY" as System
database "Base de données" as DB
participant "Service de stockage" as Storage

autonumber

' Étape 1 – Accès au formulaire
Partenaire -> PartnerUI : Demander le formulaire de création d'événement
activate PartnerUI
PartnerUI --> Partenaire : Afficher le formulaire
deactivate PartnerUI

' Étapes 2-3 – Saisie des informations et ajout des médias
Partenaire -> PartnerUI : Saisir les informations principales et ajouter les médias
activate PartnerUI

' Possibilité d’annulation avant soumission (E2)
alt partenaire annule l'opération (E2)
    Partenaire -> PartnerUI : Annuler l'opération
    PartnerUI --> Partenaire : Fin de l'interaction
    deactivate PartnerUI
else partenaire soumet l'événement
    Partenaire -> PartnerUI : Soumettre l'événement
    PartnerUI -> System : Envoyer les données (informations, médias)
    activate System

    ' Étape 4 – Validation des données
    System -> System : Valider les données saisies

    alt données invalides (E1)
        System --> PartnerUI : Message d'erreur (E1)
        PartnerUI --> Partenaire : Afficher "Informations invalides"
    else données valides
        ' Téléversement des médias
        System -> Storage : Stocker les médias
        activate Storage
        Storage --> System : Confirmation de stockage
        deactivate Storage

        ' Étape 5 – Enregistrement de l'événement
        System -> DB : Enregistrer l’événement (statut : en attente de validation)
        activate DB
        DB --> System : Confirmation d'enregistrement
        deactivate DB

        ' Étape 6 – Confirmation
        System --> PartnerUI : Message de confirmation
        PartnerUI --> Partenaire : Afficher "Événement soumis avec succès"
    end
    deactivate System
    deactivate PartnerUI
end
@enduml
