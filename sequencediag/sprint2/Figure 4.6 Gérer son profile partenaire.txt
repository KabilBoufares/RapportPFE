@startuml
actor Partenaire
boundary "Interface partenaire" as PartnerUI
control "Back-End FESTY" as Backend
database "Base de données" as DB

autonumber

' Étape 1 – Accès au profil
Partenaire -> PartnerUI : Accéder à son profil professionnel
activate PartnerUI
PartnerUI -> Backend : Demander les informations du profil partenaire
deactivate PartnerUI

activate Backend
Backend -> DB : Récupérer les informations du profil partenaire
deactivate Backend

activate DB
DB --> Backend : Informations du profil partenaire
deactivate DB

activate Backend
Backend --> PartnerUI : Envoyer les informations du profil partenaire
deactivate Backend

activate PartnerUI
PartnerUI --> Partenaire : Afficher les informations du profil
deactivate PartnerUI

' Alternative après affichage : quitter ou modifier
alt partenaire quitte la page (E2)
    Partenaire -> PartnerUI : Quitter la page sans enregistrer
    activate PartnerUI
    PartnerUI --> Partenaire : Fin de l'interaction
    deactivate PartnerUI
else partenaire modifie ses informations
    ' Étape 3 – Saisie des modifications
    Partenaire -> PartnerUI : Modifier les informations souhaitées
    activate PartnerUI
    Partenaire -> PartnerUI : Soumettre les modifications
    PartnerUI -> Backend : Envoyer les nouvelles données
    deactivate PartnerUI

    activate Backend
    Backend -> Backend : Valider les données saisies

    alt données invalides (E1)
        Backend --> PartnerUI : Message d'erreur (E1)
        deactivate Backend
        activate PartnerUI
        PartnerUI --> Partenaire : Afficher "Données invalides"
        deactivate PartnerUI
    else données valides
        Backend -> DB : Mettre à jour les informations du profil partenaire
        deactivate Backend
        activate DB
        DB --> Backend : Confirmation de mise à jour
        deactivate DB
        activate Backend
        Backend --> PartnerUI : Message de confirmation
        deactivate Backend
        activate PartnerUI
        PartnerUI --> Partenaire : Afficher "Profil mis à jour avec succès"
        deactivate PartnerUI
    end
end
@enduml
