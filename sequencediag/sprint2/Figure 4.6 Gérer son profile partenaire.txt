@startuml
actor Partenaire
participant "Interface partenaire" as PartnerUI
participant "Back-End FESTY" as System
database "Base de données" as DB

autonumber

' Étape 1 – Accès au profil
Partenaire -> PartnerUI : Accéder à son profil professionnel
activate PartnerUI
PartnerUI -> System : Demander les informations du profil partenaire
activate System
System -> DB : Récupérer les informations du profil partenaire
activate DB
DB --> System : Informations du profil partenaire
deactivate DB
System --> PartnerUI : Envoyer les informations du profil partenaire
deactivate System
PartnerUI --> Partenaire : Afficher les informations du profil
deactivate PartnerUI

' Alternative après affichage : quitter ou modifier
alt partenaire quitte la page (E2)
    Partenaire -> PartnerUI : Quitter la page sans enregistrer
    activate PartnerUI
    PartnerUI --> Partenaire : Fin de l'interaction
    deactivate PartnerUI
else partenaire modifie ses informations
    ' Étape 3 – Saisie des modifications
    Partenaire -> PartnerUI : Modifier les informations souhaitées
    activate PartnerUI
    Partenaire -> PartnerUI : Soumettre les modifications
    PartnerUI -> System : Envoyer les nouvelles données
    activate System

    ' Étape 4 – Vérification
    System -> System : Valider les données saisies

    alt données invalides (E1)
        System --> PartnerUI : Message d'erreur (E1)
        PartnerUI --> Partenaire : Afficher "Données invalides"
    else données valides
        ' Étape 5 – Enregistrement
        System -> DB : Mettre à jour les informations du profil partenaire
        activate DB
        DB --> System : Confirmation de mise à jour
        deactivate DB
        System --> PartnerUI : Message de confirmation
        PartnerUI --> Partenaire : Afficher "Profil mis à jour avec succès"
    end
    deactivate System
    deactivate PartnerUI
end
@enduml
