@startuml
actor Utilisateur
participant "Application mobile" as App
participant "Back-End FESTY" as System
database "Base de données" as DB
participant "Service de stockage" as Storage

autonumber

' Étape 1 – Accès au formulaire
Utilisateur -> App : Demander le formulaire de demande partenaire
activate App
App --> Utilisateur : Afficher le formulaire
deactivate App

' Étapes 2-3 – Saisie des informations et ajout des documents
Utilisateur -> App : Saisir les informations professionnelles et joindre les documents
activate App
App -> System : Envoyer la demande (informations, documents)
activate System

' Étape 4 – Vérification des informations
System -> System : Valider les informations saisies

alt informations incomplètes ou invalides (E1)
    System --> App : Message d'erreur (E1)
    App --> Utilisateur : Afficher "Informations invalides"
else informations valides
    ' Étape 3 (suite) – Téléversement des documents
    System -> Storage : Téléverser les documents
    activate Storage

    alt échec du téléversement (E2)
        Storage --> System : Échec du téléversement
        deactivate Storage
        System --> App : Message d'erreur (E2)
        App --> Utilisateur : Afficher "Échec du téléversement d'un document"
    else téléversement réussi
        Storage --> System : Confirmation de stockage
        deactivate Storage

        ' Étape 5 – Enregistrement de la demande
        System -> DB : Enregistrer la demande partenaire (statut : en attente)
        activate DB
        DB --> System : Confirmation
        deactivate DB

        ' Étape 6 – Confirmation
        System --> App : Confirmation de la demande
        App --> Utilisateur : Afficher "Demande enregistrée avec succès"
    end
end

deactivate System
deactivate App
@enduml
