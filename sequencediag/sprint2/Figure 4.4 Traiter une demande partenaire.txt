@startuml
actor Administrateur
participant "Interface d'administration" as AdminUI
participant "Back-End FESTY" as System
database "Base de données" as DB

autonumber

' Étape 1 – Consultation de la liste des demandes
Administrateur -> AdminUI : Demander la liste des demandes partenaires
activate AdminUI
AdminUI -> System : Demander la liste des demandes
activate System
System -> DB : Récupérer les demandes partenaires
activate DB
DB --> System : Liste des demandes
deactivate DB
System --> AdminUI : Envoyer la liste
deactivate System
AdminUI --> Administrateur : Afficher la liste des demandes
deactivate AdminUI

' Étape 2 – Sélection d'une demande
Administrateur -> AdminUI : Sélectionner une demande à traiter
activate AdminUI
AdminUI -> System : Demander le détail de la demande
activate System
System -> DB : Récupérer les informations et documents
activate DB
DB --> System : Détail de la demande
deactivate DB
System --> AdminUI : Envoyer le détail
deactivate System
AdminUI --> Administrateur : Afficher les informations et documents
deactivate AdminUI

' Étape 4 – Vérification du dossier par l'administrateur
Administrateur -> AdminUI : Examiner le dossier

alt dossier incomplet (E1)
    ' L'administrateur demande des informations complémentaires
    Administrateur -> AdminUI : Demander des informations complémentaires
    activate AdminUI
    AdminUI -> System : Signaler la demande de compléments
    activate System
    System -> DB : Mettre à jour le statut (en attente de compléments)
    activate DB
    DB --> System : Confirmation
    deactivate DB
    System --> AdminUI : Confirmation
    deactivate System
    AdminUI --> Administrateur : Afficher "Demande de compléments envoyée"
    deactivate AdminUI
else dossier complet
    ' Étape 5 – Décision (accepter ou refuser)
    Administrateur -> AdminUI : Accepter ou refuser la demande
    activate AdminUI
    AdminUI -> System : Envoyer la décision (acceptée/refusée)
    activate System
    System -> DB : Mettre à jour le statut (acceptée/refusée)
    activate DB
    DB --> System : Confirmation
    deactivate DB
    System --> AdminUI : Confirmation
    deactivate System
    AdminUI --> Administrateur : Afficher "Statut mis à jour"
    deactivate AdminUI
end
@enduml
